🔥 核心要点（2026 版）

Long-term Memory 是 Agent 系统的核心模块，用于存储历史经验、领域知识、用户偏好，突破 LLM 上下文窗口限制。

评测维度:

  1. 信息提取 (Information Extraction): 从对话中准确提取关键信息
  2. 主动推断 (Proactive Inference): 根据历史记忆主动推断用户需求
  3. 时间戳认知 (Temporal Awareness): 理解时间顺序和事件先后关系
  4. 事件认知 (Episodic Cognition): 将分散的信息组织为连贯的事件

关键趋势 (2026):

  • ✅ 托管式 Memory 服务 (Cloudflare Agent Memory, 腾讯云 Agent Memory)
  • ✅ 多模态 Memory (文本 + 图像 + 音频)
  • ✅ Graph + Vector 混合检索 (关系推理 + 语义相似度)
  • ✅ 自动记忆压缩 (LLM 定期摘要旧记忆，节省存储空间)

3. 记忆提取 (Memory Extraction)

3.1 从对话中提取关键信息

目标: 将非结构化对话转换为结构化记忆

实现方式: 使用 LLM 提取

from anthropic import Anthropic

client = Anthropic()

def extract_memory_from_conversation(conversation: list[dict]) -> list[dict]:
    """从对话中提取结构化记忆"""
    prompt = f"""
    从以下对话中提取关键信息，输出 JSON 格式：
    
    对话：
    {conversation}
    
    提取要求：
    1. 识别实体（人、地点、项目、概念）
    2. 识别关系（实体之间的关联）
    3. 识别事件（发生的时间、地点、参与者）
    4. 识别用户偏好（喜欢的工具、技术栈、学习方式）
    
    输出格式：
    {{
        "entities": [
            {{"type": "Person", "name": "张三", "attributes": {{...}}}}
        ],
        "relations": [
            {{"from": "张三", "to": "Agent 项目", "type": "参与"}}
        ],
        "events": [
            {{"time": "2026-07-01", "description": "讨论了 Agent 架构设计"}}
        ],
        "preferences": [
            {{"topic": "编程工具", "value": "偏好使用 VS Code"}}
        ]
    }}
    """
    
    response = client.messages.create(
        model="claude-sonnet-5",
        messages=[{"role": "user", "content": prompt}]
    )
    
    import json
    memory = json.loads(response.content[0].text)
    return memory

3.2 记忆提取优化技巧

技巧 1: 分块提取（避免上下文过长）

def extract_memory_in_chunks(conversation: list[dict], chunk_size: int = 10):
    """分块提取记忆"""
    all_memories = []
    
    for i in range(0, len(conversation), chunk_size):
        chunk = conversation[i:i+chunk_size]
        memory = extract_memory_from_conversation(chunk)
        all_memories.append(memory)
    
    # 合并记忆（去重）
    merged_memory = merge_memories(all_memories)
    return merged_memory

技巧 2: 增量提取（只提取新消息）

class IncrementalMemoryExtractor:
    def __init__(self):
        self.last_processed_id = None
    
    def extract_new_memories(self, new_messages: list[dict]):
        """只提取新消息的记忆"""
        # 1. 过滤已处理的消息
        if self.last_processed_id:
            new_messages = [m for m in new_messages if m["id"] > self.last_processed_id]
        
        # 2. 提取记忆
        memory = extract_memory_from_conversation(new_messages)
        
        # 3. 更新 last_processed_id
        if new_messages:
            self.last_processed_id = new_messages[-1]["id"]
        
        return memory

def store_memory_to_both(content: str, entities: list, relations: list):
    """同时存储到向量数据库和知识图谱"""
    # 1. 存储到向量数据库（用于语义检索）
    store_memory(content, metadata)
    
    # 2. 存储到知识图谱（用于关系推理）
    with neo4j_driver.session() as session:
        # 创建节点
        for entity in entities:
            session.run(
                "MERGE (e:{type} {{name: $name}}) SET e += $attributes".format(type=entity["type"]),
                name=entity["name"],
                attributes=entity["attributes"]
            )
        
        # 创建关系
        for rel in relations:
            session.run(
                """
                MATCH (a {{name: $from_name}})
                MATCH (b {{name: $to_name}})
                MERGE (a)-[r:{rel_type}]->(b)
                SET r += $attributes
                """.format(rel_type=rel["type"]),
                from_name=rel["from"],
                to_name=rel["to"],
                attributes=rel["attributes"]
            )

5. 记忆检索 (Memory Retrieval)

5.1 语义检索（Vector Similarity）

def retrieve_memory_semantic(query: str, top_k: int = 5) -> list[dict]:
    """语义检索记忆"""
    # 1. 生成查询向量
    query_embedding = embeddings.embed_query(query)
    
    # 2. 在 Milvus 中搜索
    search_params = {"metric_type": "L2", "params": {"nprobe": 10}}
    results = collection.search(
        data=[query_embedding],
        anns_field="embedding",
        param=search_params,
        limit=top_k,
        expr=None  # 可加过滤条件，如 user_id == "user_001"
    )
    
    # 3. 解析结果
    memories = []
    for hit in results[0]:
        memories.append({
            "id": hit.id,
            "content": hit.entity.get("content"),
            "timestamp": hit.entity.get("timestamp"),
            "score": hit.distance
        })
    
    return memories

5.2 图谱查询（Cypher）

def retrieve_memory_graph(query: str) -> list[dict]:
    """图谱查询记忆"""
    # 1. 将自然语言转换为 Cypher（使用 LLM）
    cypher = natural_language_to_cypher(query)
    
    # 2. 在 Neo4j 中执行查询
    with neo4j_driver.session() as session:
        results = session.run(cypher)
        
        memories = []
        for record in results:
            memories.append(dict(record))
        
        return memories

5.3 混合检索（Hybrid Search）

核心思想: 结合语义检索和图谱查询的优势

def retrieve_memory_hybrid(query: str, top_k: int = 5) -> list[dict]:
    """混合检索记忆"""
    # 1. 语义检索（向量数据库）
    semantic_results = retrieve_memory_semantic(query, top_k)
    
    # 2. 图谱查询（知识图谱）
    graph_results = retrieve_memory_graph(query)
    
    # 3. 合并结果（去重 + 排序）
    merged = merge_and_rank(semantic_results, graph_results)
    
    return merged

def merge_and_rank(semantic_results: list, graph_results: list) -> list:
    """合并和排序结果"""
    # 1. 去重（基于 content 的相似度）
    all_results = semantic_results + graph_results
    deduplicated = deduplicate_by_similarity(all_results, threshold=0.9)
    
    # 2. 排序（基于相关性分数）
    deduplicated.sort(key=lambda x: x["score"], reverse=True)
    
    return deduplicated

6. 记忆更新 (Memory Update)

6.1 记忆压缩（Memory Compression）

问题: 长期积累的记忦会占用大量存储空间，且检索效率下降

解决方案: 定期使用 LLM 压缩旧记忆

def compress_old_memories(user_id: str, days_threshold: int = 30):
    """压缩 N 天前的记忆"""
    # 1. 查询旧记忆
    old_memories = query_old_memories(user_id, days_threshold)
    
    if len(old_memories) < 10:

    记忆片段：
    {old_memories}
    
    要求：
    1. 保留关键信息（实体、关系、事件、偏好）
    2. 去除冗余信息

    """
    
    response = client.messages.create(

6.2 记忆合并（Memory Merging）

问题: 同一实体的信息可能分散在多条记忆中

解决方案: 合并相关记忆

def merge_related_memories(memory_id: str):
    """合并相关记忆"""
    # 1. 查询相关记忆（基于向量相似度）
    memory = get_memory_by_id(memory_id)
    related = retrieve_memory_semantic(memory["content"], top_k=10)
    
    # 2. 使用 LLM 合并
    prompt = f"""
    以下记忆都涉及相同或相关的信息，请合并为一条完整的记忆：
    
    记忆 1：{memory["content"]}
    记忆 2：{related[0]["content"]}
    记忆 3：{related[1]["content"]}
    ...
    
    要求：
    1. 去除重复信息
    2. 解决冲突（保留最新信息）
    3. 保留所有唯一信息
    """
    
    response = client.messages.create(
        model="claude-sonnet-5",
        messages=[{"role": "user", "content": prompt}]
    )
    
    merged_content = response.content[0].text
    
    # 3. 更新数据库
    update_memory(memory_id, merged_content)
    delete_memories([r["id"] for r in related])

6.3 记忆遗忘（Memory Forgetting）

问题: 过时信息会干扰检索结果

解决方案: 基于时间 + 相关性的遗忘策略

def forget_obsolete_memories(user_id: str):
    """遗忘过时记忆"""
    # 1. 查询所有记忆
    all_memories = query_all_memories(user_id)
    
    for memory in all_memories:
        # 2. 计算遗忘分数
        forget_score = calculate_forget_score(memory)
        
        # 3. 如果分数超过阈值，删除记忆
        if forget_score > 0.8:
            delete_memory(memory["id"])

def calculate_forget_score(memory: dict) -> float:
    """计算遗忘分数（越高越应该遗忘）"""
    score = 0.0
    
    # 因素 1: 时间衰减（越旧越容易遗忘）
    age_days = (time.time() - memory["timestamp"]) / 86400
    score += min(age_days / 365, 0.5)  # 最多贡献 0.5
    
    # 因素 2: 访问频率（越少访问越容易遗忘）
    access_count = memory.get("access_count", 0)
    score += max(1.0 - access_count / 10, 0.0) * 0.3  # 最多贡献 0.3
    
    # 因素 3: 相关性（与最近对话的相关性）
    relevance = calculate_relevance_to_recent(memory)
    score += (1.0 - relevance) * 0.2  # 最多贡献 0.2
    
    return min(score, 1.0)

def retrieve_memory_with_cache(query: str, user_id: str) -> list[dict]:
    """带缓存的记忆检索"""
    # 1. 检查缓存

    redis_client.setex(cache_key, 3600, json.dumps(memories))
    
    return memories

from prometheus_client import Counter, Histogram, Gauge
import time

# 定义指标
memory_extraction_counter = Counter("memory_extraction_total", "Total memory extractions")
memory_retrieval_latency = Histogram("memory_retrieval_latency_seconds", "Memory retrieval latency")
memory_storage_size = Gauge("memory_storage_size_bytes", "Memory storage size")

def monitor_memory_extraction(func):
    """记忆提取监控装饰器"""
    def wrapper(*args, **kwargs):
        memory_extraction_counter.inc()
        return func(*args, **kwargs)
    return wrapper

def monitor_memory_retrieval(func):
    """记忆检索监控装饰器"""
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        latency = time.time() - start
        memory_retrieval_latency.observe(latency)
        return result
    return wrapper

9. 最佳实践与常见陷阱

9.1 最佳实践

✅ Do:

  1. 使用混合检索（向量 + 图谱 + 关键词）提升召回率
  2. 定期压缩记忆（避免存储爆炸）
  3. 实现记忆遗忘（删除过时信息）
  4. 缓存热门记忆（降低检索延迟）
  5. 监控关键指标（检索延迟、存储空间、准确率）

9.2 常见陷阱

❌ Don't:

  1. 不要存储原始对话（占用空间大，检索效率低）→ 应提取结构化信息
  2. 不要忽略时间衰减（旧记忆应降低权重）→ 实现遗忘机制
  3. 不要只依赖向量检索（无法处理关系推理）→ 结合知识图谱
  4. 不要忘记隐私保护（敏感信息应加密存储）→ 使用 AES-256 加密
