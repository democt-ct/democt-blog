1. MCP 协议核心概念

1.1 什么是 MCP？

定义: MCP (Model Context Protocol) 是一个 标准化 AI 工具调用协议，用于连接 LLM 应用（Client）和工具提供方（Server）。

核心目标:

  1. 解耦: LLM 应用不直接调用工具，通过 MCP 协议间接调用
  2. 标准化: 统一的工具描述格式（JSON Schema）
  3. 安全: 内置 OAuth 2.1 认证 + 权限控制
  4. 可扩展: 支持工具动态注册、版本管理、负载均衡

1.2 MCP 架构

┌─────────────────────────────────────────────────────────────┐
│                    MCP Client (AI 应用)                    │
│  (Claude Code, Cursor, VS Code, OpenClaw)                │
└────────────────┬──────────────────────────────────────────┘
                 │ MCP Protocol (JSON-RPC 2.0)
                 │ Transport: stdio / SSE / WebSocket
                 ↓
┌─────────────────────────────────────────────────────────────┐
│                   MCP Server (工具提供方)                   │
│  Tools (可调用的函数) │ Resources (可读取的数据源)          │
│  Prompts (预定义的提示模板)                                │
└─────────────────────────────────────────────────────────────┘

三种 Transport 方式:

Transport 适用场景 优点 缺点
stdio 本地开发、测试 简单、低延迟 仅支持单机
SSE (Server-Sent Events) 生产环境、跨网络 支持流式响应、HTTP 兼容 需要 Web 服务器
WebSocket 实时双向通信 低延迟、双向通信 实现复杂

2.2 第一个 MCP Server（Python）

目标: 创建一个简单的计算器 MCP Server

# simple_calculator.py
from mcp.server import Server
from mcp.server.stdio import stdio_server
from mcp.types import Tool, TextContent
import asyncio

# 1. 创建 Server 实例
server = Server("simple-calculator")

# 2. 定义工具列表
@server.list_tools()
async def list_tools() -> list[Tool]:
    return [
        Tool(
            name="add",
            description="两个数字相加",
            inputSchema={
                "type": "object",
                "properties": {
                    "a": {"type": "number", "description": "第一个数字"},
                    "b": {"type": "number", "description": "第二个数字"}
                },
                "required": ["a", "b"]
            }
        ),
        Tool(
            name="subtract",
            description="两个数字相减",
            inputSchema={
                "type": "object",
                "properties": {
                    "a": {"type": "number", "description": "被减数"},
                    "b": {"type": "number", "description": "减数"}
                },
                "required": ["a", "b"]
            }
        )
    ]

# 3. 实现工具调用
@server.call_tool()
async def call_tool(name: str, arguments: dict) -> list[TextContent]:
    if name == "add":
        result = arguments["a"] + arguments["b"]
        return [TextContent(type="text", text=str(result))]
    elif name == "subtract":
        result = arguments["a"] - arguments["b"]
        return [TextContent(type="text", text=str(result))]
    else:
        raise ValueError(f"Unknown tool: {name}")

# 4. 启动 Server（使用 stdio transport）
async def main():
    async with stdio_server() as (read_stream, write_stream):
        await server.run(
            read_stream,
            write_stream,
            server.create_initialization_options()
        )

if __name__ == "__main__":
    asyncio.run(main())

3.1 基于 FastAPI 的 MCP Server（SSE Transport）

适用场景: 生产环境、需要跨网络访问、需要认证授权

# production_mcp_server.py
from fastapi import FastAPI, HTTPException, Depends
from mcp.server import Server
from mcp.server.sse import SseServerTransport
from mcp.types import Tool, TextContent, Resource
from contextlib import asynccontextmanager
import asyncio
import uvicorn

# 1. 创建 FastAPI 应用
app = FastAPI(title="Production MCP Server")

# 2. 创建 MCP Server 实例
mcp_server = Server("production-mcp-server")

# 3. 定义工具
@mcp_server.list_tools()
async def list_tools() -> list[Tool]:
    return [
        Tool(
            name="query_database",
            description="查询数据库",
            inputSchema={
                "type": "object",
                "properties": {
                    "sql": {"type": "string", "description": "SQL 查询语句"}
                },
                "required": ["sql"]
            }
        ),
        Tool(
            name="send_email",
            description="发送邮件",
            inputSchema={
                "type": "object",
                "properties": {
                    "to": {"type": "string", "description": "收件人"},
                    "subject": {"type": "string", "description": "主题"},
                    "body": {"type": "string", "description": "正文"}
                },
                "required": ["to", "subject", "body"]
            }
        )
    ]

# 4. 实现工具调用（带权限检查）
@mcp_server.call_tool()
async def call_tool(name: str, arguments: dict, context: dict) -> list[TextContent]:
    # 从 context 中获取用户信息（由 OAuth 2.1 提供）
    user = context.get("user")
    if not user:
        raise PermissionError("未认证的用户")
    
    if name == "query_database":
        # 检查权限：只有 admin 可以查询数据库
        if user["role"] != "admin":
            raise PermissionError("权限不足：需要 admin 角色")
        
        sql = arguments["sql"]
        # 执行 SQL 查询（示例代码）
        result = execute_sql(sql)
        return [TextContent(type="text", text=str(result))]
    
    elif name == "send_email":
        # 检查权限：所有认证用户都可以发邮件
        to = arguments["to"]
        subject = arguments["subject"]
        body = arguments["body"]
        
        # 发送邮件（示例代码）
        send_email(to, subject, body)
        return [TextContent(type="text", text="邮件已发送")]
    
    else:
        raise ValueError(f"Unknown tool: {name}")

# 5. 添加 OAuth 2.1 认证（使用 FastAPI Security）
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
security = HTTPBearer()

async def get_current_user(credentials: HTTPAuthorizationCredentials = Depends(security)):
    # 验证 OAuth 2.1 token
    token = credentials.credentials
    user = validate_oauth_token(token)
    if not user:
        raise HTTPException(status_code=401, detail="Invalid token")
    return user

# 6. 集成 MCP Server 到 FastAPI
@app.post("/mcp")
async def mcp_endpoint(request: dict, user=Depends(get_current_user)):
    # 将用户信息注入到 MCP context
    context = {"user": user}
    
    # 处理 MCP 请求
    response = await mcp_server.handle_request(request, context)
    return response

# 7. 添加 SSE Transport 支持（用于流式响应）
sse_transport = SseServerTransport(app, "/mcp/sse")

@asynccontextmanager
async def lifespan(app: FastAPI):
    # 启动时执行
    async with sse_transport:
        yield

app.router.lifespan_context = lifespan

# 8. 启动 Server
if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=8000)

3.2 生产环境关键功能

功能 1: 工具调用超时管理

import asyncio
from functools import wraps

def with_timeout(timeout_seconds: int):
    """工具调用超时装饰器"""
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            try:
                return await asyncio.wait_for(func(*args, **kwargs), timeout=timeout_seconds)
            except asyncio.TimeoutError:
                raise TimeoutError(f"工具调用超时（{timeout_seconds} 秒）")
        return wrapper
    return decorator

@mcp_server.call_tool()
@with_timeout(30)  # 30 秒超时
async def call_tool(name: str, arguments: dict):
    if name == "slow_tool":
        # 模拟慢速工具
        await asyncio.sleep(60)
        return [TextContent(type="text", text="完成")]

功能 2: 错误传播与优雅处理

from mcp.types import CallToolResult, TextContent, ErrorContent

@mcp_server.call_tool()
async def call_tool(name: str, arguments: dict) -> CallToolResult:
    try:
        result = await execute_tool(name, arguments)
        return CallToolResult(
            content=[TextContent(type="text", text=str(result))],
            isError=False
        )
    except ValueError as e:
        # 可恢复错误 → 返回错误信息，但不中断对话
        return CallToolResult(
            content=[ErrorContent(type="error", text=str(e))],
            isError=True
        )
    except Exception as e:
        # 不可恢复错误 → 抛出异常，中断对话
        raise RuntimeError(f"工具执行失败：{e}")

功能 3: 可观测性（OpenTelemetry）

4. 高级主题

4.1 动态工具注册

场景: 工具列表在运行时动态变化（如插件系统）

class DynamicMCP Server:
    def __init__(self):
        self.tools = {}
        self.server = Server("dynamic-mcp-server")
    
    def register_tool(self, name: str, description: str, input_schema: dict, handler):
        """动态注册工具"""
        tool = Tool(name=name, description=description, inputSchema=input_schema)
        self.tools[name] = {"tool": tool, "handler": handler}
        
        # 通知 Client 工具列表已更新（MCP 协议扩展）
        asyncio.create_task(self.notify_tools_changed())
    
    async def list_tools(self):
        return [t["tool"] for t in self.tools.values()]
    
    async def call_tool(self, name: str, arguments: dict):
        if name not in self.tools:
            raise ValueError(f"Tool not found: {name}")
        handler = self.tools[name]["handler"]
        return await handler(arguments)

# 使用示例
server = DynamicMCP Server()
server.register_tool(
    name="hello",
    description="打印 Hello",
    input_schema={"type": "object", "properties": {}},
    handler=lambda args: [TextContent(type="text", text="Hello!")]
)

Client 端版本选择策略:

# Client 端代码
async def select_best_tool(tools: list[Tool], tool_name: str) -> Tool:
    """选择最新版本的工具"""
    candidates = [t for t in tools if t.name.startswith(tool_name)]
    if not candidates:
        raise ValueError(f"Tool not found: {tool_name}")
    
    # 按版本号排序（语义化版本）
    candidates.sort(key=lambda t: t.annotations.get("version", "0.0"), reverse=True)
    
    # 优先选择非废弃版本
    for tool in candidates:
        if not tool.annotations.get("deprecated", False):
            return tool
    
    # 如果所有版本都废弃了，返回最新版本
    return candidates[0]

async def cross_server_workflow():

5. 实战案例：构建"Agent 知识库 MCP Server"

5.1 需求分析

目标: 为 Agent 知识库系统构建一个 MCP Server，提供以下工具：

  1. search_knowledge_base: 搜索知识库（向量检索 + 图谱检索）
  2. add_document: 添加文档到知识库
  3. update_document: 更新已有文档
  4. generate_summary: 为文档生成摘要

5.2 完整实现

# agent_knowledge_mcp_server.py
from fastapi import FastAPI
from mcp.server import Server
from mcp.types import Tool, TextContent, Resource
from pydantic import BaseModel
import asyncio
import uvicorn

# 1. 创建 Server
app = FastAPI(title="Agent Knowledge MCP Server")
mcp_server = Server("agent-knowledge-mcp")

# 2. 定义数据模型
class Document(BaseModel):
    id: str
    title: str
    content: str
    tags: list[str]

# 3. 模拟知识库（生产环境应使用向量数据库 + 图数据库）
knowledge_base = {}

# 4. 定义工具
@mcp_server.list_tools()
async def list_tools() -> list[Tool]:
    return [
        Tool(
            name="search_knowledge_base",
            description="搜索知识库",
            inputSchema={
                "type": "object",
                "properties": {
                    "query": {"type": "string", "description": "搜索关键词"},
                    "top_k": {"type": "integer", "description": "返回结果数量", "default": 5}
                },
                "required": ["query"]
            }
        ),
        Tool(
            name="add_document",
            description="添加文档到知识库",
            inputSchema={
                "type": "object",
                "properties": {
                    "title": {"type": "string", "description": "文档标题"},
                    "content": {"type": "string", "description": "文档内容"},
                    "tags": {"type": "array", "items": {"type": "string"}, "description": "标签"}
                },
                "required": ["title", "content"]
            }
        ),
        Tool(
            name="generate_summary",
            description="为文档生成摘要",
            inputSchema={
                "type": "object",
                "properties": {
                    "document_id": {"type": "string", "description": "文档 ID"}
                },
                "required": ["document_id"]
            }
        )
    ]

# 5. 实现工具调用
@mcp_server.call_tool()
async def call_tool(name: str, arguments: dict) -> list[TextContent]:
    if name == "search_knowledge_base":
        query = arguments["query"]
        top_k = arguments.get("top_k", 5)
        
        # 简单关键词匹配（生产环境应使用向量检索）
        results = []
        for doc_id, doc in knowledge_base.items():
            if query.lower() in doc["content"].lower():
                results.append(doc)
                if len(results) >= top_k:
                    break
        
        return [TextContent(type="text", text=str(results))]
    
    elif name == "add_document":
        import uuid
        doc_id = str(uuid.uuid4())
        document = {
            "id": doc_id,
            "title": arguments["title"],
            "content": arguments["content"],
            "tags": arguments.get("tags", [])
        }
        knowledge_base[doc_id] = document
        return [TextContent(type="text", text=f"文档已添加，ID: {doc_id}")]
    
    elif name == "generate_summary":
        doc_id = arguments["document_id"]
        if doc_id not in knowledge_base:
            raise ValueError(f"文档不存在: {doc_id}")
        
        document = knowledge_base[doc_id"]
        
        # 调用 LLM 生成摘要（示例代码）
        summary = await call_llm_for_summary(document["content"])
        return [TextContent(type="text", text=summary)]
    
    else:
        raise ValueError(f"Unknown tool: {name}")

# 6. 添加资源（Resources）
@mcp_server.list_resources()
async def list_resources() -> list[Resource]:
    return [
        Resource(
            uri="knowledge://documents",
            name="所有文档",
            description="知识库中的所有文档列表"
        )
    ]

@mcp_server.read_resource()
async def read_resource(uri: str) -> str:
    if uri == "knowledge://documents":
        return str(knowledge_base)
    else:
        raise ValueError(f"Unknown resource: {uri}")

    uvicorn.run(app, host="0.0.0.0", port=8000)

7. 最佳实践与常见陷阱

7.1 最佳实践

✅ Do:

  1. 使用 FastAPI + SSE Transport 用于生产环境
  2. 实现工具超时管理（避免工具执行时间过长）
  3. 添加可观测性（OpenTelemetry + Prometheus）
  4. 使用 OAuth 2.1 认证（不要自己实现认证系统）
  5. 编写单元测试 + 集成测试
  6. 使用 MCP Inspector 调试

7.2 常见陷阱

❌ Don't:

  1. 不要在生产环境使用 stdio transport（仅适用于本地开发）
  2. 不要忽略错误处理（工具调用失败应返回 isError=True，而非抛出异常）
  3. 不要硬编码敏感信息（使用环境变量或 Secret Manager）
  4. 不要在一个 Server 中注册过多工具（建议 < 50 个工具，过多会影响性能）
  5. 不要忘记版本管理（工具接口变更时，使用版本号区分）
